      requiresAuth: true,
      requiredPermissions: ['customers.read'],
      handler: async ({ req, user }) => {
        - Parse query params with customerQuerySchema
        - Find user's internal companyId from JWT
        - Build Drizzle query on customers table:
          - WHERE companyId = user's companyId AND deletedAt IS NULL
          - If search: add OR condition on name ILIKE, email ILIKE, phone ILIKE
          - If tag_id: join with customer_tags junction table
          - ORDER BY sort_by column (default: name)
          - LIMIT and OFFSET for pagination
        - Get total count with same WHERE conditions
        - Return paginatedResponse with { data, meta: { total, page, limit, total_pages } }
        - IMPORTANT: Map internal IDs to UUIDs in response (uuid field, not id)
      }
    })
