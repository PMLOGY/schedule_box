    ```sql
    CREATE OR REPLACE FUNCTION current_user_role() RETURNS TEXT AS $$
    ```

    - customer_tags: No company_id. Skip — accessed through customer.
    - employee_services: No company_id. Skip — accessed through employee.
    - service_resources: No company_id. Skip — accessed through service.
    - role_permissions: Global. Skip.

