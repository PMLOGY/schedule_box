    **Step 3: Rewrite BookingCalendar.tsx** (`apps/web/components/booking/BookingCalendar.tsx`)

    Replace the FullCalendar implementation with react-big-calendar. The new component must:

    a) **Import structure:**
    ```typescript
    'use client';
    import { useMemo, useState, useCallback } from 'react';
    import { Calendar, dateFnsLocalizer, Views } from 'react-big-calendar';
    import withDragAndDrop from 'react-big-calendar/lib/addons/dragAndDrop';
    import { format, parse, startOfWeek, getDay } from 'date-fns';
    import { cs } from 'date-fns/locale/cs';
    import 'react-big-calendar/lib/css/react-big-calendar.css';
    import 'react-big-calendar/lib/addons/dragAndDrop/styles.css';
    import { useQuery } from '@tanstack/react-query';
    import { useCalendarStore } from '@/stores/calendar.store';
    import { useRescheduleBooking } from '@/hooks/use-reschedule-booking';
    import { apiClient } from '@/lib/api-client';
    import type { Booking, PaginatedResponse } from '@schedulebox/shared/types';
    import BookingDetailPanel from './BookingDetailPanel';
    import '../../styles/calendar.css';
    ```

    b) **Set up the date-fns localizer:**
    ```typescript
    const locales = { cs };
    const localizer = dateFnsLocalizer({ format, parse, startOfWeek, getDay, locales });
    ```

    c) **Create the DnD-enhanced Calendar:**
    ```typescript
    const DnDCalendar = withDragAndDrop(Calendar);
    ```

    d) **Event mapping:** Transform bookings from the API into react-big-calendar event format:
    ```typescript
    interface CalendarEvent {
      id: string;
      title: string;
      start: Date;
      end: Date;
      resource?: string;
      booking: Booking;
      isDraggable: boolean;
    }
    ```

    Map from `Booking` to `CalendarEvent`:
    - `id`: `String(booking.id)`
    - `title`: `${booking.customer?.name ?? ''} - ${booking.service?.name ?? ''}`
    - `start`: `new Date(booking.startTime)`
    - `end`: `new Date(booking.endTime)`
    - `booking`: the full booking object (for detail panel)
    - `isDraggable`: `['pending', 'confirmed'].includes(booking.status)`

    e) **Use React Query for data fetching** (replacing FullCalendar's event source callback):
    ```typescript
    // Calculate date range from the current view
    const { view, selectedDate, selectedEmployeeIds, showCancelled } = useCalendarStore();

    // Compute visible range based on view
    const { dateFrom, dateTo } = useMemo(() => {
      // For day: selectedDate to selectedDate
      // For week: start of week to end of week
      // For month: start of month to end of month (with padding)
      // Use date-fns functions
    }, [view, selectedDate]);

    const { data: bookingsData } = useQuery({
      queryKey: ['bookings', 'calendar', dateFrom, dateTo],
      queryFn: () => apiClient.get<PaginatedResponse<Booking>>('/bookings', {
        date_from: dateFrom,
        date_to: dateTo,
        limit: 200,
      }),
    });
    ```

    f) **Filter events client-side:**
    - If `!showCancelled`, filter out `status === 'cancelled'`
    - If `selectedEmployeeIds.length > 0`, filter to matching employees

    g) **Handle drag-drop via `onEventDrop`:**
    ```typescript
    const rescheduleMutation = useRescheduleBooking();

    const handleEventDrop = useCallback(({ event, start }: { event: CalendarEvent; start: Date }) => {
      if (!event.isDraggable) return;
      rescheduleMutation.mutate({
        bookingId: event.id,
        startTime: start.toISOString(),
      });
    }, [rescheduleMutation]);
    ```
    Note: react-big-calendar does not have a `revert` function like FullCalendar. The React Query cache invalidation in `useRescheduleBooking.onSettled` will refetch and correct the UI. For immediate feedback, the `onError` handler already shows a toast.

    h) **Handle event click:**
    ```typescript
    const [selectedBookingId, setSelectedBookingId] = useState<string | null>(null);
    const [detailPanelOpen, setDetailPanelOpen] = useState(false);

    const handleSelectEvent = useCallback((event: CalendarEvent) => {
      setSelectedBookingId(event.id);
      setDetailPanelOpen(true);
    }, []);
    ```

    i) **Event styling via `eventPropGetter`:**
    ```typescript
    const STATUS_COLORS: Record<string, string> = {
      pending: '#F59E0B',
      confirmed: '#3B82F6',
      completed: '#10B981',
      cancelled: '#9CA3AF',
      no_show: '#EF4444',
    };

    const eventPropGetter = useCallback((event: CalendarEvent) => ({
      style: {
        backgroundColor: STATUS_COLORS[event.booking.status] ?? '#3B82F6',
        borderRadius: '0.375rem',
        border: 'none',
        fontSize: '0.875rem',
        padding: '2px 4px',
      },
    }), []);
    ```

    j) **Render the DnDCalendar:**
    ```tsx
    <div className="relative rounded-lg border bg-card p-4">
      <DnDCalendar
        localizer={localizer}
        events={calendarEvents}
        view={view}
        date={selectedDate}
        onView={(newView) => useCalendarStore.getState().setView(newView as CalendarView)}
        onNavigate={(date) => useCalendarStore.getState().setSelectedDate(date)}
        onEventDrop={handleEventDrop}
        onSelectEvent={handleSelectEvent}
        eventPropGetter={eventPropGetter}
        draggableAccessor={(event) => (event as CalendarEvent).isDraggable}
        min={new Date(0, 0, 0, 6, 0, 0)}   // 06:00
        max={new Date(0, 0, 0, 22, 0, 0)}  // 22:00
        step={15}
        timeslots={4}
        toolbar={false}
        culture="cs"
        style={{ minHeight: 600 }}
        formats={{
          timeGutterFormat: 'HH:mm',
          eventTimeRangeFormat: ({ start, end }) =>
            `${format(start, 'HH:mm')} - ${format(end, 'HH:mm')}`,
        }}
      />
    </div>
    ```

    k) **Keep the BookingDetailPanel** at the bottom, same as current implementation.

    **Important:** Do NOT remove the FullCalendar packages from package.json yet — `calendar-view.tsx` (the resource timeline view in the dashboard) still uses them. That component is separate and not being modified in this task.
