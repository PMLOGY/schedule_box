    1. Run `cd apps/web && pnpm dlx storybook@8 init --type nextjs --no-open` to scaffold Storybook. This installs @storybook/nextjs, @storybook/addon-essentials, and storybook as devDependencies in apps/web/package.json.

    3. Configure .storybook/main.ts:
    ```typescript
    import type { StorybookConfig } from '@storybook/nextjs';
    const config: StorybookConfig = {
      stories: ['../apps/web/**/*.stories.@(ts|tsx)'],
      addons: ['@storybook/addon-essentials'],
      framework: { name: '@storybook/nextjs', options: {} },
      docs: { autodocs: 'tag' },
    };
    export default config;
    ```

    4. Configure .storybook/preview.ts:
    ```typescript
    import type { Preview } from '@storybook/react';
    import '../apps/web/app/globals.css';
    const preview: Preview = {
      parameters: {
        backgrounds: {
          default: 'gradient',
          values: [
            { name: 'gradient', value: 'linear-gradient(135deg, #0f172a 0%, #1e1b4b 50%, #0f172a 100%)' },
            { name: 'light', value: '#f8fafc' },
          ],
        },
      },
    };
    export default preview;
    ```
